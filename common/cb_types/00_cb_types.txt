		has_reform = religious_permanent_revolution_reform

			AND  = {
				OR = {
					cb_on_primitives = yes
					cb_on_overseas = yes

			}
		}
		NOT = { religion = FROM }

	}

	war_goal = superiority_world_crusade
}

# Force migration
cb_force_migration = {

	valid_for_subject = no

	prerequisites = {
		has_dlc = "Conquest of Paradise"
		OR = {
			is_native_tribe = yes
			is_migratory_tribe = yes
		}
		FROM = {
			is_migratory_tribe = yes
			OR = {
				AND = {
					is_native_tribe = yes
					capital_scope = {
						tribal_land_of = ROOT
					}
				}
				is_native_tribe = no
			}
		}
	}
	war_goal = force_migration
}

cb_native_american_tribal_feud = {
	valid_for_subject = no

	prerequisites = {
		has_dlc = "Conquest of Paradise"
		is_native_tribe = yes
		FROM = {
			is_native_tribe = yes
			if = {
				limit = { 
					capital_scope = {
						tribal_land_of = FROM
					}
				}
				is_neighbor_of = ROOT # This is faster method so let's do this for the general case
			}
			else = {
				any_tribal_land = {
					any_neighbor_province = {
						tribal_land_of = ROOT
					}
				}
			}
		}
	}
	war_goal = take_native_american_land

	attacker_disabled_po = {
		po_become_vassal
		po_subjugate_vassal
		po_become_tributary_state
		po_subjugate_tributary_state
	}
}
