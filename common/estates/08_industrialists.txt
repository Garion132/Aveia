estate_industrialists ={
	icon = 3

	# If true, country will get estate
	trigger = {	
		has_country_flag = industrialists_active
	}

	# These scale with loyalty & power
	country_modifier_happy = {
		global_trade_goods_size_modifier = 0.1
		development_cost = -0.1
	}
	country_modifier_neutral = {
		global_trade_goods_size_modifier = 0.1
	}	
	country_modifier_angry = {
		global_trade_goods_size_modifier = -0.1
		development_cost = 0.1
		global_unrest = 2
	}
	land_ownership_modifier = {
		industrialists_loyalty_modifier = 0.2
	}
	
	province_independence_weight = {
		factor = 1
		modifier = {
			factor = 0.1
			NOT = { base_production = 3 }
		}
		modifier = {
			factor = 0.5
			OR = {
				NOT = { base_production = 6 }
			}
		}
		modifier = {
			factor = 2
			base_production = 10
		}
		modifier = {
			factor = 3
			OR = {
				trade_goods = coal
				#trade_goods = manufactured_goods
				trade_goods = machinery
			}
		}
		modifier = {
			factor = 0.5
			is_territorial_core = owner
		}
		modifier = {
			factor = 1.5
			has_building = artisans_guild
		}
		modifier = {
			factor = 2
			has_building = craftsmans_district
		}
		modifier = {
			factor = 3
			has_building = industrial_zone
		}
	}

	# Influence modifiers
	base_influence = 10
	influence_modifier = {	
		desc = EST_HIGH_DEVELOPMENT_PROVINCE
		trigger = {
			any_owned_province = {
				development = 30
			}
		}
		influence = 5
	}
	influence_modifier = {
		desc = EST_VAL_LOW_PRODUCTION_INCOME
		trigger = {
			NOT = { production_income_percentage = 0.25 }
		}	
		influence = -5
	}
	influence_modifier = {
		desc = EST_VAL_HIGH_PRODUCTION_INCOME
		trigger = {
			production_income_percentage = 0.5
			NOT = { production_income_percentage = 0.75 }
		}	
		influence = 5
	}
	influence_modifier = {
		desc = EST_VAL_VERY_HIGH_PRODUCTION_INCOME
		trigger = {
			production_income_percentage = 0.75
		}	
		influence = 10
	}
	influence_modifier = {
		desc = EST_VAL_OTHER_ESTATE_IN_POWER
		trigger = {
			OR = {
				has_disaster = estate_nobility_disaster
				has_disaster = estate_church_disaster
				has_disaster = estate_burghers_disaster
			}
		}	
		influence = -40
	}
	
	loyalty_modifier = {
		desc = EST_VAL_OTHER_ESTATE_IN_POWER_LOY
		trigger = {
			OR = {
				has_disaster = estate_nobility_disaster
				has_disaster = estate_church_disaster
				has_disaster = estate_burghers_disaster
			}
		}
		loyalty = -20
	}

	color = { 35 35 150 }

	privileges = {
		estate_industrialists_land_rights
	}

	agendas = {

	}

	influence_from_dev_modifier = 1.5
}
