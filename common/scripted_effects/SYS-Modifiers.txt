#Visible Pop Modifier Code
update_pop_modifiers = {
	clear_pop_modifiers = yes
	if = {
		limit = { check_variable = { which = total_pop value = 8 } }
		if = {
			limit = { check_variable = { which = total_pop value = 32 } }
			if = {
				limit = { check_variable = { which = total_pop value = 128 } }
				if = {
					limit = { check_variable = { which = total_pop value = 256 } }
					add_permanent_province_modifier = { name = pop_modifier_9 duration = -1 } # 256K
				}
				else = {
					add_permanent_province_modifier = { name = pop_modifier_8 duration = -1 } # 128K
				}
			}
			else = {
				if = {
					limit = { check_variable = { which = total_pop value = 64 } }
					add_permanent_province_modifier = { name = pop_modifier_7 duration = -1 } # 64K
				}
				else = {
					add_permanent_province_modifier = { name = pop_modifier_6 duration = -1 } # 32K
				}
			}
		}
		else = {
			if = {
				limit = { check_variable = { which = total_pop value = 16 } }
				add_permanent_province_modifier = { name = pop_modifier_5 duration = -1 } # 16K
			}
			else = {
				add_permanent_province_modifier = { name = pop_modifier_4 duration = -1 } # 8K
			}
		}
	}
	else = {
		if = {
			limit = { check_variable = { which = total_pop value = 2 } }
			if = {
				limit = { check_variable = { which = total_pop value = 4 } }
				add_permanent_province_modifier = { name = pop_modifier_3 duration = -1 } # 4K
			}
			else = {
				add_permanent_province_modifier = { name = pop_modifier_2 duration = -1 } # 2K
			}
		}
		else = {
			if = {
				limit = { check_variable = { which = total_pop value = 1 } }
				add_permanent_province_modifier = { name = pop_modifier_1 duration = -1 } # 1K
			}
			else = {
				add_permanent_province_modifier = { name = pop_modifier_0 duration = -1 } #<1K
			}
		}
	}
}
