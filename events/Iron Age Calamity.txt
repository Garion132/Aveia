country_event = {
	id = iron_age_calamity.3
	title = iron_age_calamity.3.t
	desc = iron_age_calamity.3.d
	picture = COMET_SIGHTED_eventPicture

	is_triggered_only = yes

    immediate = {
		set_global_flag = iron_age_calamity_eruption
		if = {
			limit = {
				NOT = {
					OR = {
						technology_group = northern_men_group
						technology_group = free_fauns_group
						capital_scope = {
							OR={
								continent = elteria
								continent = haqsili
							}
						}
					}
				}
			}
			set_country_flag = iron_age_calamity_possible
			hidden_effect = {
				iac_add_calamity_modifiers = yes
			}
		}
	}

	option = {
		name = iron_age_calamity.3.opt.a #Aveia Trembles
		trigger = {
			capital_scope = { NOT = { superregion = aveia_superregion } }
		}
		add_stability = -1
		add_legitimacy = -20
		add_republican_tradition = -10
		add_war_exhaustion = 3

		if = {
			limit = {
				total_development = 20
				NOT = { total_development = 50 }
			}
			iac_add_country_devastation = { all = yes base_value = 5 }
		}
		if = {
			limit = {
				total_development = 50
				NOT = { total_development = 200 }
			}
			iac_add_country_devastation = { all = yes base_value = 7 }
		}
		if = {
			limit = {
				total_development = 200
			}

		}

			limit = {
				total_development = 50
				NOT = { total_development = 200 }
			}
			iac_add_country_devastation = { all = yes base_value = 15 }
		}
		if = {
			limit = {
				total_development = 200
			}
			iac_add_country_devastation = { all = yes base_value = 20 }
		}

		custom_tooltip = orbury_volcano_damage_tt
		if = {
			limit ={
				NOT = { has_global_flag = orbury_volcano_damage }
			}
			set_global_flag = orbury_volcano_damage
