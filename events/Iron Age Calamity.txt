	desc = "iron_age_calamity.103.d"
	picture = ANGRY_MOB_eventPicture

		any_owned_province = {
			is_capital = no

			}
			is_city = yes
		}
	}

	is_triggered_only = yes

		days = 1
	}

		hidden_effect = {

		}
	}
