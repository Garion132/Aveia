	title = eastern_scramble.25.t
	desc = eastern_scramble.25.d
	
	#is_triggered_only = yes
	
	trigger={
		ai = yes
		num_of_colonists = 1
		NOT={total_development = 50}
		NOT={num_of_colonies = 1}
		OR={
			capital_scope = {
				OR={
					region = qirdud_region
					region = anoric_region
					region = rogosian_heights_region
					region = vortingrad_region
					region = windsweaps_region
				}
			}
			is_subject_of_type = colony_elteria_start
			tag = HIR
		}
		
	}
	mean_time_to_happen={
		months = 2
	}

	option = {
		name = eastern_scramble.25.a
		random_owned_province = { 
			random_empty_neighbor_province = {
				create_colony = 100
			}
		}
	}
}
