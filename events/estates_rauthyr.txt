#Low loyalty - etailia
country_event = {
	id = rauthyr_estate_events.1
	title = rauthyr_estate_events.1.t
	desc = rauthyr_estate_events.1.d
	picture = NOBLE_ESTATE_DEMANDS_eventPicture
	
	trigger = {
		NOT={dynasty = "Etailia"}
		has_estate = estate_house_etailia
		NOT = {
			estate_loyalty = {
				estate = estate_house_etailia
				loyalty = 30
			}
		}
		estate_territory = { 
			estate = estate_house_etailia
			territory = 5
		}
		NOT = {
			has_estate_influence_modifier = {
				estate = estate_house_etailia
				modifier = EST_VAL_NOBLES_EXTORTED_RIGHTS
			}
		}
		NOT = {
			has_estate_influence_modifier = {
				estate = estate_house_etailia
				modifier = EST_VAL_NOBLES_DENIED_RIGHTS
			}
		}
		any_owned_province = {
			is_territory = no
			is_capital = no
			is_city = yes
		}
	}
	
	is_triggered_only = yes
	
	immediate = {
		hidden_effect = {
			while = {
				limit = {
					NOT = { 
						development_in_provinces = {
							has_province_flag = house_etailia_estate_revolt
							value = estate
							estate = estate_house_etailia
						}
					}
					any_owned_province = {
						is_territory = no
						is_capital = no
						is_city = yes
						NOT = { has_province_flag = house_etailia_estate_revolt }
					}
					check_reducing_estate_revolt_size_trigger = {
						FLAG = house_etailia_estate_revolt
					}
				}
				random_owned_province = {
					limit = {
						is_territory = no
						is_capital = no
						is_city = yes
						NOT = { has_province_flag = house_etailia_estate_revolt }
					}
					set_province_flag = house_etailia_estate_revolt
				}
			}
		}
	}
	
	mean_time_to_happen = {
		days = 1
	}
	
	option = { 
		name = rauthyr_estate_events.1.a #Let it be.
		ai_chance = {
			factor = 1
		}
		add_estate_influence_modifier = {
			estate = estate_house_etailia
			desc = EST_VAL_NOBLES_EXTORTED_RIGHTS
			influence = 10
			duration = 5475
		}
		every_owned_province = {
			limit = { has_province_flag = house_etailia_estate_revolt }
			add_local_autonomy = 33
		}
		hidden_effect = {
			every_owned_province = {
				limit = { has_province_flag = house_etailia_estate_revolt }
				clr_province_flag = house_etailia_estate_revolt
			}
		}
	}
	option = {
		name = rauthyr_estate_events.1.b #
		ai_chance = {
			factor = 0
		}
		add_estate_influence_modifier = {
			estate = estate_house_etailia
			desc = EST_VAL_NOBLES_DENIED_RIGHTS
			influence = -10
			duration = 5475
		}
		every_owned_province = {
			limit = { has_province_flag = house_etailia_estate_revolt }
			clr_province_flag = house_etailia_estate_revolt
			spawn_rebels = { #Always possible in noble province.
				type = etailia_pretender_rebels
				size = 1
				leader_dynasty = "Etailia"
			}
		}
	}
}

country_event = {
	id = rauthyr_estate_events.7
	title = rauthyr_estate_events.7.t
	desc = rauthyr_estate_events.7.d
	picture = NOBLE_ESTATE_DEMANDS_eventPicture
	
	trigger = {
		tag = RAU
		OR={
			has_heir = no
			has_female_heir = yes
			has_foreign_heir = yes
			NOT={legitimacy_equivalent = 50}
		}
		ruler_age = 35
		#NOT={has_ruler_flag = RAU_heir_confirmed}
		NOT={has_heir_flag = RAU_heir_confirmed}
	}
	
	mean_time_to_happen = {
		days = 720
	}
	
	option = { #if heir is female, and has legitimacy, can burn legitimacy and house loyalty to keep
		name = rauthyr_estate_events.7.a #Let it be.
		trigger={
			has_heir = yes
			legitimacy_equivalent = 80
		}
		ai_chance = {
			factor = 1
		}
		set_heir_flag = RAU_heir_confirmed
		add_estate_loyalty = {
			estate = estate_house_kethai
			loyalty = -5
		}
		add_estate_loyalty = {
			estate = estate_house_etailia
			loyalty = -5
		}
		add_estate_loyalty = {
			estate = estate_house_nyletmi
			loyalty = -5
		}
